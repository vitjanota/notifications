Example 4

Notifications with structured content using custom templates

Initialization

let notifications3 = new Notifications(
    {},
    {
        error: `<h1>My Error Notification: %%title%%</h1>
        <div data-process="body">
            <span>%%content%%</span>
            <a href="%%src%%" data-type="link">%%content%%</a>
        </div>`,
        mytest: `<h1>%%header%%</h1>
        <ul data-process="list">
            <li>%%content%%</li>
        </ul>`
    }
);

Function call

notifications.emit([
    {
        title: 'message example',
        body: [
            {content:'This is a '},
            {
                src: '#anchor',
                content: 'link',
                type: 'link'
            },
            {content: '.'}
        ]
    }
],'error');
notifications.emit([
    {
        header: 'my custom message',
        list: [
            {content:'item 1'},
            {content: 'item 2'}
        ]
    }
],'mytest');
